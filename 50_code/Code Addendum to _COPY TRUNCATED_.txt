def coupled\_oscillator\_hamiltonian(n\_max, omega, lambda\_coupling):

    """Create Hamiltonian matrix for two coupled oscillators."""

    dim = (n\_max + 1) \*\* 2

    H = np.zeros((dim, dim))

    

    for n\_A in range(n\_max + 1):

        for n\_B in range(n\_max + 1):

            idx = n\_A \* (n\_max + 1) + n\_B

            

            \# Diagonal: ω(n\_A + n\_B + 1)

            H\[idx, idx\] = omega \* (n\_A + n\_B + 1)

            

            \# Off-diagonal: coupling λ x\_A x\_B

            \# x = sqrt(ħ/2mω)(a + a†), so x\_A x\_B couples (n\_A, n\_B) to nearby states

            for dn\_A, dn\_B in \[(-1, -1), (-1, 1), (1, -1), (1, 1)\]:

                n\_A\_new, n\_B\_new = n\_A + dn\_A, n\_B + dn\_B

                if 0 \<= n\_A\_new \<= n\_max and 0 \<= n\_B\_new \<= n\_max:

                    idx\_new = n\_A\_new \* (n\_max + 1) + n\_B\_new

                    matrix\_element = lambda\_coupling \* 0.5 \* (

                        np.sqrt(max(n\_A, n\_A\_new)) \* np.sqrt(max(n\_B, n\_B\_new))

                    )

                    H\[idx, idx\_new\] += matrix\_element

    

    return H

  
  

def compute\_purity(psi, n\_max):

    """Compute purity of subsystem A."""

    dim\_A = n\_max + 1

    dim\_B = n\_max + 1

    

    \# Reshape to matrix form

    psi\_matrix = psi.reshape(dim\_A, dim\_B)

    

    \# Reduced density matrix

    rho\_A = psi\_matrix @ psi\_matrix.conj().T

    

    \# Purity

    gamma = np.real(np.trace(rho\_A @ rho\_A))

    return gamma

  
  

def simulate\_entanglement\_dynamics(n\_max, omega, lambda\_coupling, t\_max, n\_steps):

    """Simulate coupled oscillators and track purity."""

    H = coupled\_oscillator\_hamiltonian(n\_max, omega, lambda\_coupling)

    

    \# Initial state: |α⟩\_A ⊗ |0⟩\_B (coherent state in A, vacuum in B)

    alpha = 2.0  \# coherent state amplitude

    psi\_A = np.zeros(n\_max + 1, dtype=complex)

    for n in range(n\_max + 1):

        psi\_A\[n\] = np.exp(-abs(alpha)\*\*2 / 2) \* alpha\*\*n / np.sqrt(np.math.factorial(n))

    psi\_B = np.zeros(n\_max + 1, dtype=complex)

    psi\_B\[0\] = 1.0

    

    psi\_0 = np.outer(psi\_A, psi\_B).flatten()

    psi\_0 /= np.linalg.norm(psi\_0)

    

    \# Time evolution

    times = np.linspace(0, t\_max, n\_steps)

    dt = times\[1\] - times\[0\]

    U\_dt = expm(-1j \* H \* dt)

    

    purity\_trajectory = \[\]

    psi = psi\_0.copy()

    

    for t in times:

        gamma = compute\_purity(psi, n\_max)

        purity\_trajectory.append(gamma)

        psi = U\_dt @ psi

    

    return times, np.array(purity\_trajectory)

  
  

def compute\_geodesic\_deviation(x\_trajectory, u\_trajectory):

    """Measure deviation from geodesic in hyperbolic metric."""

    \# Convert to hyperbolic angle

    psi = np.arctanh(np.sqrt(1 - x\_trajectory\*\*2))  \# since u = tanh²ψ

    

    \# For a geodesic, d²ψ/dt² + Γ (dψ/dt)² = 0

    \# Simplified test: check if ψ(t) is approximately linear or follows known geodesic form

    

    \# Fit to polynomial

    t = np.arange(len(psi))

    p1 = np.polyfit(t, psi, 1)  \# linear fit

    p2 = np.polyfit(t, psi, 2)  \# quadratic fit

    

    residual\_linear = np.std(psi - np.polyval(p1, t))

    residual\_quadratic = np.std(psi - np.polyval(p2, t))

    

    return {

        'linear\_residual': residual\_linear,

        'quadratic\_residual': residual\_quadratic,

        'is\_geodesic\_like': residual\_linear \< 0.1 \* np.std(psi)

    }

def partition\_function(beta, E0, psi\_max=20):

    """Compute Z(β) numerically."""

    def integrand(psi):

        return np.exp(-beta \* E0 \* np.cosh(psi)) / np.cosh(psi)

    Z, \_ = quad(integrand, 0, psi\_max)

    return Z

  
  

def compute\_thermodynamics(beta\_array, E0):

    """Compute Z, \<E\>, C\_V, S for array of β values."""

    results = {'beta': beta\_array, 'Z': \[\], 'E': \[\], 'C\_V': \[\], 'S': \[\]}

    

    for beta in beta\_array:

        \# Partition function

        Z = partition\_function(beta, E0)

        results\['Z'\].append(Z)

        

        \# \<E\> via numerical derivative of ln Z

        dbeta = beta \* 1e-5

        Z\_plus = partition\_function(beta + dbeta, E0)

        Z\_minus = partition\_function(beta - dbeta, E0)

        dlogZ\_dbeta = (np.log(Z\_plus) - np.log(Z\_minus)) / (2 \* dbeta)

        E\_avg = -dlogZ\_dbeta

        results\['E'\].append(E\_avg)

        

        \# C\_V via second derivative

        d2logZ\_dbeta2 = (np.log(Z\_plus) - 2\*np.log(Z) + np.log(Z\_minus)) / dbeta\*\*2

        C\_V = -beta\*\*2 \* d2logZ\_dbeta2  \# in units of k\_B

        results\['C\_V'\].append(C\_V)

        

        \# Entropy

        S = np.log(Z) + beta \* E\_avg  \# in units of k\_B

        results\['S'\].append(S)

    

    return {k: np.array(v) for k, v in results.items()}

    Z\_analytic = kn(0, beta \* E0)  \# K\_0 Bessel function

    return Z\_numerical, Z\_analytic, np.abs(Z\_numerical - Z\_analytic) / Z\_analytic

def identify\_boundary\_interior(N, boundary\_width=1):

    """Return indices of boundary and interior sites."""

    all\_indices = np.arange(N \* N).reshape(N, N)

    boundary\_mask = np.zeros((N, N), dtype=bool)

    boundary\_mask\[:boundary\_width, :\] = True

    boundary\_mask\[-boundary\_width:, :\] = True

    boundary\_mask\[:, :boundary\_width\] = True

    boundary\_mask\[:, -boundary\_width:\] = True

    

    boundary\_idx = all\_indices\[boundary\_mask\].flatten()

    interior\_idx = all\_indices\[\~boundary\_mask\].flatten()

    return boundary\_idx, interior\_idx

  
  

def compute\_schur\_complement(L, boundary\_idx, interior\_idx):

    """Compute the DtN/Schur complement Λ = L\_BB - L\_BI L\_II⁻¹ L\_IB."""

    L\_BB = L\[np.ix\_(boundary\_idx, boundary\_idx)\].toarray()

    L\_BI = L\[np.ix\_(boundary\_idx, interior\_idx)\].toarray()

    L\_IB = L\[np.ix\_(interior\_idx, boundary\_idx)\].toarray()

    L\_II = L\[np.ix\_(interior\_idx, interior\_idx)\].toarray()

    

    \# Schur complement

    Lambda = L\_BB - L\_BI @ np.linalg.solve(L\_II, L\_IB)

    return Lambda

def hopf\_map(z1, z2):

    """Map S³ → S² via Hopf fibration."""

    x = np.abs(z1)\*\*2 - np.abs(z2)\*\*2

    y = 2 \* np.real(z1 \* np.conj(z2))

    z = 2 \* np.imag(z1 \* np.conj(z2))

    return np.array(\[x, y, z\])

  
  

def compute\_berry\_phase(hamiltonian\_path, n\_steps=1000):

    """

    Compute Berry phase for adiabatic evolution around a closed loop

    in parameter space.

    """

    phases = \[\]

    psi\_prev = None

    

    for i, H in enumerate(hamiltonian\_path):

        \# Ground state

        eigenvalues, eigenvectors = np.linalg.eigh(H)

        psi = eigenvectors\[:, 0\]  \# ground state

        

        if psi\_prev is not None:

            \# Fix gauge: make overlap with previous state real and positive

            overlap = np.vdot(psi\_prev, psi)

            psi \*= np.exp(-1j \* np.angle(overlap))

        

        if i \> 0:

            \# Accumulate phase

            phase = np.angle(np.vdot(psi\_prev, psi))

            phases.append(phase)

        

        psi\_prev = psi.copy()

    

    \# Final phase to close the loop

    H\_final = hamiltonian\_path\[0\]

    \_, eigenvectors = np.linalg.eigh(H\_final)

    psi\_final = eigenvectors\[:, 0\]

    overlap = np.vdot(psi\_prev, psi\_final)

    psi\_final \*= np.exp(-1j \* np.angle(overlap))

    phases.append(np.angle(np.vdot(psi\_prev, psi\_final)))

    

    total\_phase = np.sum(phases)

    return total\_phase

def test\_geodesic\_time\_bound():

    """

    Test TEP Prediction 2: Δt\_min = d(γ₁,γ₂) / √(2P)

    

    Strategy: For many initial and final purity values,

    search for the fastest quantum evolution connecting them.

    Compare to geodesic prediction.

    """

    def geodesic\_distance(g1, g2):

        """TEP geodesic distance between purity states."""

        if g1 \<= 0 or g2 \<= 0 or g1 \>= 1 or g2 \>= 1:

            return np.inf

        return 0.5 \* np.abs(np.log(g2 \* (1-g1) / (g1 \* (1-g2))))

    

    def find\_fastest\_transition(gamma\_target, tol=0.01, max\_tries=1000):

        """

        Search for Hamiltonian that reaches gamma\_target fastest.

        Start from pure state (γ=1).

        """

        dim\_A, dim\_B = 2, 2

        dim = dim\_A \* dim\_B

        

        best\_time = np.inf

        best\_H = None

        

        for \_ in range(max\_tries):

            \# Random Hamiltonian, normalized

            H = np.random.randn(dim, dim) + 1j \* np.random.randn(dim, dim)

            H = (H + H.conj().T) / 2

            H = H / np.linalg.norm(H, ord=2)

            

            \# Start from product pure state (γ=1)

            psi = np.zeros(dim, dtype=complex)

            psi\[0\] = 1.0

            

            \# Binary search for time to reach target purity

            t\_low, t\_high = 0.0, 100.0

            

            for \_ in range(50):  \# Binary search iterations

                t\_mid = (t\_low + t\_high) / 2

                U = expm(-1j \* H \* t\_mid)

                psi\_t = U @ psi

                rho = np.outer(psi\_t, psi\_t.conj())

                rho\_A = np.trace(rho.reshape(dim\_A, dim\_B, dim\_A, dim\_B),

                    if t\_mid \< best\_time:

                        best\_time = t\_mid

                        best\_H = H

                    break

        

        return best\_time, best\_H

            continue

            

        \# Theoretical maximum ratio

        R\_max = np.sqrt(2 \* (2 \* gamma\_target - 1)) / gamma\_target

        

        \# Empirical bound value

        emp\_bound = 2 \* gamma\_target \* np.sqrt(1 - gamma\_target)

        violation\_count = 0

        near\_optimal\_count = 0

            \# γ = λ² + (1-λ)², solve for λ given γ

            discriminant = 2 \* gamma\_target - 1

                continue

            lam = (1 + np.sqrt(discriminant)) / 2

            U\_A = np.linalg.qr(np.random.randn(2,2) + 1j\*np.random.randn(2,2))\[0\]

            U\_B = np.linalg.qr(np.random.randn(2,2) + 1j\*np.random.randn(2,2))\[0\]

            H = np.random.randn(dim, dim) + 1j \* np.random.randn(dim, dim)

def analyze\_approach\_to\_equilibrium(times, purities):

    """

    Fit the approach to γ = 0.5 with both models:

    1. Exponential: γ - 0.5 = A exp(-Γt)

    2. Power-law: γ - 0.5 = B (t\* - t)²

    """

    gamma\_eq = 0.5

    y = purities - gamma\_eq

    

    \# Only fit where y \> 0.01 (away from noise floor)

    mask = y \> 0.01

    t\_fit = times\[mask\]

    y\_fit = y\[mask\]

    

    if len(t\_fit) \< 10:

        return None

    

    \# Exponential fit: log(y) = log(A) - Γt

    log\_y = np.log(y\_fit)

    exp\_fit = np.polyfit(t\_fit, log\_y, 1)

    Gamma\_exp = -exp\_fit\[0\]

    A\_exp = np.exp(exp\_fit\[1\])

    y\_exp\_pred = A\_exp \* np.exp(-Gamma\_exp \* t\_fit)

    exp\_residual = np.sum((y\_fit - y\_exp\_pred)\*\*2)

    

    \# Power-law fit: √y = √B (t\* - t), so √y vs t is linear

    sqrt\_y = np.sqrt(y\_fit)

    pow\_fit = np.polyfit(t\_fit, sqrt\_y, 1)

    \# sqrt\_y = a + b\*t → t\* = -a/b, √B = -b\*t\* = a

    if pow\_fit\[0\] \< 0:  \# Slope should be negative

        t\_star = -pow\_fit\[1\] / pow\_fit\[0\]

        B\_pow = pow\_fit\[1\]\*\*2

        y\_pow\_pred = np.maximum(0, pow\_fit\[1\] + pow\_fit\[0\] \* t\_fit)\*\*2

        pow\_residual = np.sum((y\_fit - y\_pow\_pred)\*\*2)

    else:

        pow\_residual = np.inf

        t\_star = np.inf

    

    return {

        'exp\_rate': Gamma\_exp,

        'exp\_residual': exp\_residual,

        'pow\_t\_star': t\_star,

        'pow\_residual': pow\_residual,

        'times': t\_fit,

        'data': y\_fit,

        'exp\_pred': y\_exp\_pred,

        'pow\_pred': y\_pow\_pred if pow\_residual \< np.inf else None

    }

        \# Combined transfer matrix

        M = np.array(\[\[phase / t, r \* phase / t\],

                      \[r \* np.conj(phase) / t, np.conj(phase) / t\]\])

        return M

        \# Total transfer matrix

        M\_total = np.eye(2, dtype=complex)

import numpy as np

from scipy.linalg import expm

from scipy.sparse import csr\_matrix, kron as sparse\_kron, eye as sparse\_eye

from scipy.sparse.linalg import expm\_multiply

class ThermalizationTest:

    """

    Test TEP prediction: equilibration rate non-monotonic in temperature.

    

    Key prediction:

    - Standard Lindblad: Γ(T) increases with T

    - TEP: Γ(T) peaks at intermediate T, decreases toward T → ∞

    """

    

    def \_\_init\_\_(self, n\_bath=10, coupling\_strength=0.15):

        self.n\_bath = n\_bath

        self.n\_total = n\_bath + 1

        self.dim = 2 \*\* self.n\_total

        self.coupling\_strength = coupling\_strength

        

        \# Pauli matrices (sparse)

        self.sz = csr\_matrix(\[\[1, 0\], \[0, -1\]\], dtype=complex)

        self.sp = csr\_matrix(\[\[0, 1\], \[0, 0\]\], dtype=complex)

        self.sm = csr\_matrix(\[\[0, 0\], \[1, 0\]\], dtype=complex)

        self.I2 = sparse\_eye(2, dtype=complex, format='csr')

        

    def \_embed(self, op, site):

        """Embed single-site operator into full Hilbert space (sparse)."""

        result = csr\_matrix(\[\[1\]\], dtype=complex)

        for i in range(self.n\_total):

            result = sparse\_kron(result, op if i == site else self.I2, format='csr')

        return result

    

    def build\_hamiltonian(self, bath\_frequencies, couplings):

        """

        Build system-bath Hamiltonian.

        

        H = (Δ/2)σz + Σⱼ(ωⱼ/2)τzʲ + Σⱼgⱼ(σ⁺τ⁻ʲ + σ⁻τ⁺ʲ)

        """

        Delta = 1.0  \# Qubit gap (sets energy scale)

        

        H = Delta \* self.\_embed(self.sz, 0) / 2

        

        for j in range(self.n\_bath):

            H += bath\_frequencies\[j\] \* self.\_embed(self.sz, j + 1) / 2

            H += couplings\[j\] \* (

                self.\_embed(self.sp, 0) @ self.\_embed(self.sm, j + 1) +

                self.\_embed(self.sm, 0) @ self.\_embed(self.sp, j + 1)

            )

        

        return H

    

    def get\_qubit\_purity(self, psi):

        """Compute purity of qubit subsystem."""

        psi\_matrix = psi.reshape(2, 2\*\*self.n\_bath)

        rho\_q = psi\_matrix @ psi\_matrix.conj().T

        return np.real(np.trace(rho\_q @ rho\_q))

    

    def get\_equilibrium\_purity(self, T):

        """

        Equilibrium purity at temperature T.

        

        For qubit with gap Δ=1: γ\_eq = (p₀² + p₁²)

        where p₁ = 1/(1 + e^(1/T)), p₀ = 1 - p₁

        """

        if T \< 0.01:

            return 1.0  \# Ground state

        

        p\_excited = 1 / (1 + np.exp(1/T))

        p\_ground = 1 - p\_excited

        return p\_ground\*\*2 + p\_excited\*\*2

    

    def sample\_thermal\_bath\_state(self, T, bath\_frequencies):

        """Sample pure bath state from thermal ensemble."""

        n\_bath\_dim = 2 \*\* self.n\_bath

        

        if T \> 100:  \# Effectively infinite temperature

            \# Uniform random state (Haar measure approximation)

            psi\_bath = np.random.randn(n\_bath\_dim) + 1j \* np.random.randn(n\_bath\_dim)

            psi\_bath /= np.linalg.norm(psi\_bath)

        else:

            \# Compute bath energies for each configuration

            energies = np.zeros(n\_bath\_dim)

            for config in range(n\_bath\_dim):

                for j in range(self.n\_bath):

                    spin = 1 if (config \>\> j) & 1 else -1

                    energies\[config\] += bath\_frequencies\[j\] \* spin / 2

            

            \# Boltzmann weights

            beta = 1/T if T \> 0 else np.inf

            weights = np.exp(-beta \* (energies - energies.min()))

            probs = weights / weights.sum()

            

            \# Sample configuration

            config = np.random.choice(n\_bath\_dim, p=probs)

            psi\_bath = np.zeros(n\_bath\_dim, dtype=complex)

            psi\_bath\[config\] = 1.0

        

        return psi\_bath

    

    def run\_single\_trajectory(self, T, t\_max=150, dt=0.5, seed=None):

        """

        Run single thermalization trajectory.

        

        Returns: times, purities, gamma\_eq

        """

        if seed is not None:

            np.random.seed(seed)

        

        \# Random bath parameters

        bath\_frequencies = 1.0 + 0.3 \* np.random.randn(self.n\_bath)

        couplings = self.coupling\_strength \* (1 + 0.2 \* np.random.randn(self.n\_bath))

        

        \# Build Hamiltonian

        H = self.build\_hamiltonian(bath\_frequencies, couplings)

        

        \# Initial state: qubit excited, bath thermal

        psi\_bath = self.sample\_thermal\_bath\_state(T, bath\_frequencies)

        psi\_qubit = np.array(\[0, 1\], dtype=complex)  \# |1⟩ = excited

        psi = np.kron(psi\_qubit, psi\_bath)

        

        \# Time evolution using Krylov method

        times = np.arange(0, t\_max, dt)

        purities = np.zeros(len(times))

        

        for i, t in enumerate(times):

            purities\[i\] = self.get\_qubit\_purity(psi)

            if i \< len(times) - 1:

                psi = expm\_multiply(-1j \* H \* dt, psi)

        

        gamma\_eq = self.get\_equilibrium\_purity(T)

        

        return times, purities, gamma\_eq

    

    def extract\_relaxation\_rate(self, times, purities, gamma\_eq):

        """

        Extract relaxation rate from purity decay.

        

        Fits: γ(t) - γ\_eq = A \* exp(-Γ \* t)

        

        Returns: Γ, fit\_quality (R²)

        """

        y = purities - gamma\_eq

        

        \# Use region where signal is clear

        mask = (y \> 0.02) & (y \< 0.9 \* y\[0\])

        

        if np.sum(mask) \< 5:

            return np.nan, 0.0

        

        t\_fit = times\[mask\]

        y\_fit = y\[mask\]

        

        try:

            \# Linear fit on log scale

            log\_y = np.log(y\_fit)

            coeffs = np.polyfit(t\_fit, log\_y, 1)

            gamma = -coeffs\[0\]

            

            \# R² calculation

            y\_pred = np.exp(coeffs\[1\] + coeffs\[0\] \* t\_fit)

            ss\_res = np.sum((y\_fit - y\_pred)\*\*2)

            ss\_tot = np.sum((y\_fit - y\_fit.mean())\*\*2)

            r\_squared = 1 - ss\_res / ss\_tot if ss\_tot \> 0 else 0

            

            return gamma, r\_squared

        except:

            return np.nan, 0.0

    def \_\_init\_\_(self, n\_bath=10, coupling\_strength=0.15):

        self.n\_bath = n\_bath

        self.n\_total = n\_bath + 1

        self.dim = 2 \*\* self.n\_total

        self.coupling\_strength = coupling\_strength

        

        self.sz = csr\_matrix(\[\[1, 0\], \[0, -1\]\], dtype=complex)

        self.sp = csr\_matrix(\[\[0, 1\], \[0, 0\]\], dtype=complex)

        self.sm = csr\_matrix(\[\[0, 0\], \[1, 0\]\], dtype=complex)

        self.I2 = sparse\_eye(2, dtype=complex, format='csr')

        

    def \_embed(self, op, site):

        result = csr\_matrix(\[\[1\]\], dtype=complex)

        for i in range(self.n\_total):

            result = sparse\_kron(result, op if i == site else self.I2, format='csr')

        return result

    

    def build\_hamiltonian(self, bath\_frequencies, couplings):

        Delta = 1.0

        H = Delta \* self.\_embed(self.sz, 0) / 2

        for j in range(self.n\_bath):

            H += bath\_frequencies\[j\] \* self.\_embed(self.sz, j + 1) / 2

            H += couplings\[j\] \* (

                self.\_embed(self.sp, 0) @ self.\_embed(self.sm, j + 1) +

                self.\_embed(self.sm, 0) @ self.\_embed(self.sp, j + 1)

            )

        return H

class FullCouplingTEPTest:

    """

    Use coupling that does NOT conserve Sz:

    H\_int = g \* (σ\_x ⊗ σ\_x + σ\_y ⊗ σ\_y + σ\_z ⊗ σ\_z) = Heisenberg

    Or simply: H\_int = g \* σ\_x ⊗ σ\_x (random field term)

    

    This allows full thermalization\!

    """

    

    def \_\_init\_\_(self, n\_bath=10, coupling\_strength=0.15, coupling\_type='heisenberg'):

        self.n\_bath = n\_bath

        self.n\_total = n\_bath + 1

        self.dim = 2 \*\* self.n\_total

        self.coupling\_strength = coupling\_strength

        self.coupling\_type = coupling\_type

        

        \# Pauli matrices

        self.sz = csr\_matrix(\[\[1, 0\], \[0, -1\]\], dtype=complex)

        self.sx = csr\_matrix(\[\[0, 1\], \[1, 0\]\], dtype=complex)

        self.sy = csr\_matrix(\[\[0, -1j\], \[1j, 0\]\], dtype=complex)

        self.sp = csr\_matrix(\[\[0, 1\], \[0, 0\]\], dtype=complex)

        self.sm = csr\_matrix(\[\[0, 0\], \[1, 0\]\], dtype=complex)

        self.I2 = sparse\_eye(2, dtype=complex, format='csr')

        

    def \_embed(self, op, site):

        """Embed single-site operator into full Hilbert space."""

        result = csr\_matrix(\[\[1\]\], dtype=complex)

        for i in range(self.n\_total):

            result = sparse\_kron(result, op if i == site else self.I2, format='csr')

        return result

    

    def build\_hamiltonian(self, bath\_frequencies, couplings):

        """Build Hamiltonian with Sz-breaking coupling."""

        Delta = 1.0  \# Qubit frequency

        

        \# Qubit term

        H = Delta \* self.\_embed(self.sz, 0) / 2

        

        for j in range(self.n\_bath):

            \# Bath terms

            H += bath\_frequencies\[j\] \* self.\_embed(self.sz, j + 1) / 2

            

            if self.coupling\_type == 'heisenberg':

                \# Full Heisenberg: conserves total S but NOT Sz

                H += couplings\[j\] \* (

                    self.\_embed(self.sx, 0) @ self.\_embed(self.sx, j + 1) +

                    self.\_embed(self.sy, 0) @ self.\_embed(self.sy, j + 1) +

                    self.\_embed(self.sz, 0) @ self.\_embed(self.sz, j + 1)

                )

            elif self.coupling\_type == 'xx\_zz':

                \# XX + ZZ: breaks Sz conservation

                H += couplings\[j\] \* (

                    self.\_embed(self.sp, 0) @ self.\_embed(self.sm, j + 1) +

                    self.\_embed(self.sm, 0) @ self.\_embed(self.sp, j + 1) +

                    0.5 \* self.\_embed(self.sz, 0) @ self.\_embed(self.sz, j + 1)

                )

            elif self.coupling\_type == 'random':

                \# Random coupling in all directions (most generic)

                H += couplings\[j\] \* self.\_embed(self.sx, 0) @ self.\_embed(self.sx, j + 1)

                H += couplings\[j\] \* 0.3 \* self.\_embed(self.sz, 0) @ self.\_embed(self.sz, j + 1)

            else:  \# 'xx' - the problematic one

                H += couplings\[j\] \* (

                    self.\_embed(self.sp, 0) @ self.\_embed(self.sm, j + 1) +

                    self.\_embed(self.sm, 0) @ self.\_embed(self.sp, j + 1)

                )

        

        return H

    

    def get\_qubit\_purity(self, psi):

        """Get purity of qubit reduced density matrix."""

        psi\_matrix = psi.reshape(2, 2\*\*self.n\_bath)

        rho\_q = psi\_matrix @ psi\_matrix.conj().T

        return np.real(np.trace(rho\_q @ rho\_q))

    

    def get\_qubit\_populations(self, psi):

        """Get (p\_ground, p\_excited) of qubit."""

        psi\_matrix = psi.reshape(2, 2\*\*self.n\_bath)

        rho\_q = psi\_matrix @ psi\_matrix.conj().T

        return np.real(rho\_q\[0, 0\]), np.real(rho\_q\[1, 1\])

    

    def thermal\_state\_energy(self, energies, T):

        """Get thermal probability distribution."""

        beta = 1/T if T \> 0 else 1000

        weights = np.exp(-beta \* (energies - energies.min()))

        return weights / weights.sum()

    

    def run\_trajectory(self, T, t\_max=150, dt=0.5, seed=None):

        """Run single trajectory and extract purity dynamics."""

        rng = np.random.default\_rng(seed)

        

        \# Bath setup

        bath\_frequencies = 1.0 + 0.5 \* rng.standard\_normal(self.n\_bath)

        couplings = self.coupling\_strength \* (1 + 0.3 \* rng.standard\_normal(self.n\_bath))

        

        H = self.build\_hamiltonian(bath\_frequencies, couplings)

        

        \# Initialize bath thermally

        n\_bath\_dim = 2 \*\* self.n\_bath

        energies = np.zeros(n\_bath\_dim)

        for config in range(n\_bath\_dim):

            for j in range(self.n\_bath):

                spin = 1 if (config \>\> j) & 1 else -1

                energies\[config\] += bath\_frequencies\[j\] \* spin / 2

        

        probs = self.thermal\_state\_energy(energies, T)

        config = rng.choice(n\_bath\_dim, p=probs)

        psi\_bath = np.zeros(n\_bath\_dim, dtype=complex)

        psi\_bath\[config\] = 1.0

        

        \# Qubit starts excited

        psi\_qubit = np.array(\[0, 1\], dtype=complex)

        psi = np.kron(psi\_qubit, psi\_bath)

        

        times = np.arange(0, t\_max, dt)

        purity = np.zeros(len(times))

        p\_ground = np.zeros(len(times))

        

        for i, t in enumerate(times):

            purity\[i\] = self.get\_qubit\_purity(psi)

            p\_ground\[i\], \_ = self.get\_qubit\_populations(psi)

            if i \< len(times) - 1:

                psi = expm\_multiply(-1j \* H \* dt, psi)

        

        return times, purity, p\_ground

class TEPValidation:

    """

    Clean TEP validation showing key physics:

    1. Trajectories approach the diagonal curve (decoherence)

    2. Equilibrium point lies ON the diagonal curve

    3. Location of equilibrium determined by energy conservation

    """

    

    def \_\_init\_\_(self, n\_bath=10, coupling\_strength=0.15):

        self.n\_bath = n\_bath

        self.n\_total = n\_bath + 1

        self.dim = 2 \*\* self.n\_total

        self.coupling\_strength = coupling\_strength

        

        \# Pauli matrices

        self.sz = csr\_matrix(\[\[1, 0\], \[0, -1\]\], dtype=complex)

        self.sp = csr\_matrix(\[\[0, 1\], \[0, 0\]\], dtype=complex)

        self.sm = csr\_matrix(\[\[0, 0\], \[1, 0\]\], dtype=complex)

        self.I2 = sparse\_eye(2, dtype=complex, format='csr')

        

    def \_embed(self, op, site):

        result = csr\_matrix(\[\[1\]\], dtype=complex)

        for i in range(self.n\_total):

            result = sparse\_kron(result, op if i == site else self.I2, format='csr')

        return result

    

    def build\_hamiltonian(self, bath\_frequencies, couplings):

        Delta = 1.0

        H = Delta \* self.\_embed(self.sz, 0) / 2

        

        for j in range(self.n\_bath):

            H += bath\_frequencies\[j\] \* self.\_embed(self.sz, j + 1) / 2

            H += couplings\[j\] \* (

                self.\_embed(self.sp, 0) @ self.\_embed(self.sm, j + 1) +

                self.\_embed(self.sm, 0) @ self.\_embed(self.sp, j + 1)

            )

        return H

    

    def get\_qubit\_observables(self, psi):

        """Extract qubit density matrix and all observables."""

        psi\_matrix = psi.reshape(2, 2\*\*self.n\_bath)

        rho\_q = psi\_matrix @ psi\_matrix.conj().T

        

        p\_e = np.real(rho\_q\[1, 1\])

        C = np.abs(rho\_q\[0, 1\])

        gamma = np.real(np.trace(rho\_q @ rho\_q))

        

        \# Entropy

        eigs = np.linalg.eigvalsh(rho\_q)

        eigs = eigs\[eigs \> 1e-15\]

        S = -np.sum(eigs \* np.log(eigs))

        

        return p\_e, C, gamma, S

    

    def run\_trajectory(self, T, t\_max=300, dt=0.5, seed=None):

        """Run and return trajectory data."""

        rng = np.random.default\_rng(seed)

        

        bath\_frequencies = 1.0 + 0.5 \* rng.standard\_normal(self.n\_bath)

        couplings = self.coupling\_strength \* (1 + 0.3 \* rng.standard\_normal(self.n\_bath))

        

        H = self.build\_hamiltonian(bath\_frequencies, couplings)

        

        \# Thermal bath

        n\_bath\_dim = 2 \*\* self.n\_bath

        energies = np.zeros(n\_bath\_dim)

        for config in range(n\_bath\_dim):

            for j in range(self.n\_bath):

                spin = 1 if (config \>\> j) & 1 else -1

                energies\[config\] += bath\_frequencies\[j\] \* spin / 2

        

        beta = 1/T if T \> 0 else 100

        probs = np.exp(-beta \* (energies - energies.min()))

        probs /= probs.sum()

        config = rng.choice(n\_bath\_dim, p=probs)

        psi\_bath = np.zeros(n\_bath\_dim, dtype=complex)

        psi\_bath\[config\] = 1.0

        

        \# Qubit excited

        psi = np.kron(np.array(\[0, 1\], dtype=complex), psi\_bath)

        

        times = np.arange(0, t\_max, dt)

        data = {'t': times, 'p\_e': \[\], 'C': \[\], 'gamma': \[\], 'S': \[\]}

        

        for i in range(len(times)):

            p\_e, C, gamma, S = self.get\_qubit\_observables(psi)

            data\['p\_e'\].append(p\_e)

            data\['C'\].append(C)

            data\['gamma'\].append(gamma)

            data\['S'\].append(S)

            

            if i \< len(times) - 1:

                psi = expm\_multiply(-1j \* H \* dt, psi)

        

        for key in \['p\_e', 'C', 'gamma', 'S'\]:

            data\[key\] = np.array(data\[key\])

        

        return data

  
  
  
  

def canonical\_curve(n\_points=500):

    """Canonical γ(S) curve for diagonal qubit states."""

    p\_vals = np.linspace(1e-8, 1-1e-8, n\_points)

    gamma\_vals = (1 - p\_vals)\*\*2 + p\_vals\*\*2

    S\_vals = -p\_vals \* np.log(p\_vals) - (1 - p\_vals) \* np.log(1 - p\_vals)

    return S\_vals, gamma\_vals, p\_vals

  
  
  
  

def thermal\_point(T, Delta=1.0):

    """Thermal equilibrium point for temperature T."""

    if T \<= 0:

        p\_e = 0.0

    else:

        beta = 1/T

        p\_e = np.exp(-beta \* Delta/2) / (np.exp(beta \* Delta/2) + np.exp(-beta \* Delta/2))

    

    gamma = (1 - p\_e)\*\*2 + p\_e\*\*2

    if p\_e \< 1e-10 or p\_e \> 1 - 1e-10:

        S = 0

    else:

        S = -p\_e \* np.log(p\_e) - (1 - p\_e) \* np.log(1 - p\_e)

    return S, gamma, p\_e

    def predict\_microcanonical\_p(T\_bath, n\_bath=10, Delta=1.0):

        """

        Predict equilibrium p\_e from energy conservation.

        

        Initial energy: E\_q = +Δ/2 (excited) + E\_bath(T)

        Final: qubit at p\_e, bath absorbs/releases energy

        

        For rough estimate: p\_e ≈ p\_thermal(T\_eff) where T\_eff accounts

        for energy injected by excited qubit.

        """

        \# Initial qubit energy (excited state)

        E\_qubit\_initial = Delta / 2

        

        \# If qubit relaxes to p\_e, it releases energy ΔE = (1 - 2\*p\_e) \* Δ/2

        \# This heats the bath slightly

        

        \# Simple estimate: effective temperature seen by qubit

        \# Bath heat capacity \~ n\_bath (for spin bath)

        \# ΔT \~ ΔE / C \~ Δ/2 / n\_bath (if qubit fully relaxes)

        

        \# More accurate: solve self-consistently

        \# p\_e should satisfy detailed balance at some effective T

        

        if T\_bath \< 0.1:

            T\_bath = 0.1  \# Avoid division issues

        beta = 1/T\_bath

        p\_thermal = 1 / (1 + np.exp(beta \* Delta))

        E\_released = (1 - 2\*p\_thermal) \* Delta / 2

        

        \# This heats bath, raising effective temperature

        \# Simple model: T\_eff = T + E\_released / (heat\_capacity)

        \# Heat capacity of n spins at temperature T

        C\_bath = n\_bath \* (Delta/2)\*\*2 / (T\_bath\*\*2 \* np.cosh(beta\*Delta/2)\*\*2)

        C\_bath = max(C\_bath, n\_bath \* 0.1)  \# Floor to avoid singularity

        

        T\_eff = T\_bath + E\_released / max(C\_bath, 1)

        T\_eff = max(T\_eff, 0.1)

        

        \# Predicted p at effective temperature

        beta\_eff = 1/T\_eff

        p\_predicted = 1 / (1 + np.exp(beta\_eff \* Delta))
